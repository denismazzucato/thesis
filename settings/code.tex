% Define colors for syntax highlighting


\definecolor{codegray}{gray}{0.5}
\definecolor{codecomment}{gray}{0.6}


\lstdefinelanguage{customPython}{
  language=Python,
  morekeywords={then,floor,function,do,entry,done,else,skip, assert}
}

% Define code listing style
\lstdefinestyle{mystyle}{
  commentstyle=\scriptsize\color{codecomment},
  % keywordstyle=\color{seabornRed},
  % make keywords just bold
  keywordstyle=\bfseries,
  numberstyle=\tiny\color{codegray},
  stringstyle=\color{seabornRed},
  basicstyle=\ttfamily\footnotesize,
  breakatwhitespace=false,
  breaklines=true,
  % captionpos=b,
  keepspaces=true,
  numbers=left,
  numbersep=5pt,
  showspaces=false,
  showstringspaces=false,
  showtabs=false,
  tabsize=2
}
